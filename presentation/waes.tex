\section{Motivation}
\begin{frame}
\frametitle{A Quick Review of VAEs}
Let our data be distributed according to $P_X$.  

\col{Variational Auto-Encoders} (VAEs) seek to find a generative model (i.e. decoder) $G$ that minimizes
\begin{align*}
D_{KL}(P_X, P_G)
\end{align*} 
\end{frame}
